#include <WiFi.h>
#include <WebServer.h>
const char* ssid = "ESP32_ROBOT";
const char* password = "12345678";
WebServer server(80);
#define IN1 14
#define IN2 27
#define IN3 33
#define IN4 32
#define ENA 26
#define ENB 25  // add second enable pin if you have it, else tie to ENA

#define PWM_FREQ       1000
#define PWM_RESOLUTION 8

int spd = 200; // default speed 0-255

void setup() {
  Serial.begin(115200);
  pinMode(IN1,OUTPUT); pinMode(IN2,OUTPUT);
  pinMode(IN3,OUTPUT); pinMode(IN4,OUTPUT);

  // PWM on enable pins
  ledcAttach(ENA, PWM_FREQ, PWM_RESOLUTION);
  ledcAttach(ENB, PWM_FREQ, PWM_RESOLUTION);

  WiFi.softAP(ssid,password);
  Serial.println("AP IP: " + WiFi.softAPIP().toString());

  server.on("/",handleRoot);
  server.on("/forward",forward);
  server.on("/backward",backward);
  server.on("/left",left);
  server.on("/right",right);
  server.on("/stop",stopCar);
  server.on("/speed",setSpeed);
  server.begin();
}
void loop(){ server.handleClient(); }

// ── Speed setter ───────────────────────────────────────────────────
void setSpeed(){
  if(server.hasArg("val"))
    spd = constrain(server.arg("val").toInt(), 0, 255);
  Serial.println("Speed: " + String(spd));
  server.send(200,"text/plain","OK");
}

// ── Motor functions ────────────────────────────────────────────────
void forward(){
  digitalWrite(IN1,HIGH); digitalWrite(IN2,LOW);
  digitalWrite(IN3,HIGH); digitalWrite(IN4,LOW);
  ledcWrite(ENA, spd);    ledcWrite(ENB, spd);
  server.send(200,"text/plain","");
}
void backward(){
  digitalWrite(IN1,LOW);  digitalWrite(IN2,HIGH);
  digitalWrite(IN3,LOW);  digitalWrite(IN4,HIGH);
  ledcWrite(ENA, spd);    ledcWrite(ENB, spd);
  server.send(200,"text/plain","");
}
void left(){
  digitalWrite(IN1,LOW);  digitalWrite(IN2,HIGH);
  digitalWrite(IN3,HIGH); digitalWrite(IN4,LOW);
  ledcWrite(ENA, spd);    ledcWrite(ENB, spd);
  server.send(200,"text/plain","");
}
void right(){
  digitalWrite(IN1,HIGH); digitalWrite(IN2,LOW);
  digitalWrite(IN3,LOW);  digitalWrite(IN4,HIGH);
  ledcWrite(ENA, spd);    ledcWrite(ENB, spd);
  server.send(200,"text/plain","");
}
void stopCar(){
  digitalWrite(IN1,LOW); digitalWrite(IN2,LOW);
  digitalWrite(IN3,LOW); digitalWrite(IN4,LOW);
  ledcWrite(ENA, 0);     ledcWrite(ENB, 0);
  server.send(200,"text/plain","");
}
